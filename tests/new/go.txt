ELEMENT                 [A-Z]+
STRING                  "([\w!#-&\(-\[\]-\}]|\\[tnr"'])*"
COLON                   :
ATTRIBUTE               [a-z]+
ADDRESS                 @[0-9]+
LBRACK                  \{
RBRACK                  \}
COMMA                   ,
DIRECTIVE               #[A-Za-z]+
ELLIPSIS                \.\.\.

%%
%ignore \w
%ignore \t
%ignore \n

%start start

start     => item { item };

item      => struct
           | directive
           ;

struct    => ELEMENT COLON LBRACK { attribute } RBRACK ;

attribute => ATTRIBUTE COLON litlist;

litlist   => ELLIPSIS
           | { literal }
           ;

literal   => feature
           | STRING
           | ADDRESS
           ;

feature   => ELEMENT [ featlist ];

featlist  => LBRACK ELEMENT { COMMA ELEMENT } RBRACK;

directive => DIRECTIVE diritem { COMMA diritem };

diritem   => STRING
           | ELEMENT
           ;
